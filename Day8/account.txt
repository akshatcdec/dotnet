        [HttpPost]
        public IActionResult ChangePassword(ChangePasswordViewModel vw)
        {

            if(vw.NewPassword != vw.RetypePassword)
            {
                ViewData["MSG"] = "New Password Not match";
                return View(vw);
            }
            else
            {
                ViewData["MSG"] = "Password Changed";
                ModelState.Clear();
                return View(null);

            }
        }
